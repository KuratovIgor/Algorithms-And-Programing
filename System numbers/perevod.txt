int charToInt(char ch)
{
	if (ch >= 65 && ch <= 90)
		return ch - 'A' + 10;
	else
		return ch - '0';
}

char intToChar(int value)
{
	if (value <= 9)
		return (char)(value + '0');
	else
		return (char)(value + 55);
}
