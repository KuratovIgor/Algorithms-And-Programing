#include <iostream>

using namespace std;


int main()
{
	setlocale(LC_ALL, "ru");

	int N;
	cout << "¬ведите колчество чисел: ";
	cin >> N;

	for (int i = 2; i < N; i++)
	{
		int count = 0;
		for (int j = 2;  j * j <= i;  j++)
		{
			if (i % j == 0)
				count++;
		}
		if (count == 0)
			cout << i << '\t';
	}

	return 0;
}
